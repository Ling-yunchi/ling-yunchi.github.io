# 高精度数

题目中有时会出现一些巨大的数，超过了所有数据类型存储的范围，很让人头疼 qwqq

高精度数便是解决该问题的方法

所谓的高精度数就是使用数组来存储每一位的数据

然后根据基本运算来计算，此时需要考虑进位的问题

原理听上去很简单，但实际上敲的话还是很麻烦的 qwqq

# c 艹 (string) 实现

c++ 的 string 确实比 c 语言的字符串好使多了

但因为我一开始不知道如何在 string 前面添加字符，所有绕了很大的弯子还做不出来 qwqq

后来发现只要加在前面就行了才知道 string 真香 (￣y▽ ￣)╭

参考: 大数运算

#include<iostream>
#include<cstdio>
#include<vector>
#include<string>
#include<cctype>
using namespace std;

int main()
{
	string num1;
	string num2;
	/*char c;
	while ((c = getchar()) && c != '\n')
		num1 += c;
	while ((c = getchar()) && c != '\n')
		num2 += c;*/
	cin >> num1 >> num2;

	int len1 = num1.length();
	int len2 = num2.length();
	//补全较短者前面的0
	if (len1 > len2)
	{
		int k = len1 - len2;
		for (int i = 0; i < k; i++)
			num2 = "0" + num2;
	}
	else if (len1 < len2)
	{
		int k = len2 - len1;
		for (int i = 0; i < k; i++)
			num1 = "0" + num1;
	}
	//从后往前计算
	string ans;
	int carry = 0;//记录进位
	for (int i = len1 > len2 ? len1 - 1 : len2 - 1; i >= 0; i--)
	{
		int temp = num1[i] - '0' + num2[i] - '0' + carry;
		carry = temp / 10;
		temp %= 10;
		ans = char(temp + '0') + ans;//强制类型转换(char)(temp+'0')
	}
	if (carry)//如果carry不等于0,前面进位
		ans = char(carry + '0') + ans;

	for (auto it = ans.begin(); it != ans.end(); it++)
		cout << *it;

	return 0;
}

# 乘法

# c 艹

原理是模拟手算进位 qwqq

#include<iostream>
#include<cstdio>
#include<vector>
#include<string>
#include<cctype>
using namespace std;
constexpr auto maxn = 20000;
char num1[maxn];
char num2[maxn];
int ans[maxn];

int main()
{
	string n1, n2;
	cin >> n1 >> n2;//读入
	//转换为数字,反向存入数组
	for (int i = 0; i < n1.length(); i++)
		num1[i] = n1[n1.length() - 1 - i] - '0';
	for (int i = 0; i < n2.length(); i++)
		num2[i] = n2[n2.length() - 1 - i] - '0';
	//计算每一位存入ans
	for (int i = 0; i < n1.length(); i++)
		for (int j = 0; j < n2.length(); j++)
			ans[i + j] += num1[i] * num2[j];
	//处理进位
	int len = n1.length() + n2.length();
	for (int i = 0; i <len; i++)
		if (ans[i] > 9)
		{
			ans[i + 1] += ans[i] / 10;
			ans[i] %= 10;
		}
	//减少输出的位数,不输出前面的0
	while (ans[len] == 0 && len > 0)len--;
	//输出结果
	for (int i = len; i >= 0; i--)
		printf("%d", ans[i]);

	return 0;
}
