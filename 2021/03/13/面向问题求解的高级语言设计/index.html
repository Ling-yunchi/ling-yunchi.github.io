# stl 中的全排列函数 next_permutation

需要某些数据的全排列进行枚举时可以使用 stl 中的 permutation 来使用数据的全排列 qwqq

  1. next_permutation：求下一个排列组合

    • 函数模板：next_permutation (arr, arr+size);
    • 参数说明：
      • arr： 数组名
      • size：数组元素个数
    • 函数功能： 返回值为 bool 类型，当当前序列不存在下一个排列时，函数返回 false，否则返回 true，排列好的数在数组中存储
    • 注意：在使用前需要对欲排列数组按升序排序，否则只能找出该序列之后的全排列数。
      • 比如，如果数组 num 初始化为 2,3,1，那么输出就变为了：{2 3 1}

  2. prev_permutation：求上一个排列组合

    • 函数模板：prev_permutation (arr, arr+size);
    • 参数说明：
      • arr： 数组名
      • size：数组元素个数
    • 函数功能： 返回值为 bool 类型，当当前序列不存在上一个排列时，函数返回 false，否则返回 true
    • 注意：在使用前需要对欲排列数组按降序排序，否则只能找出该序列之后的全排列数。

该函数是以字典序排序

# 查找

# 二分

  • 二分查找也称折半查找（Binary Search），它是一种效率较高的查找方法
  • 要求查找的线性表是有单调性的（这样可以做到排除一半）

二分答案：

在答案所在的范围内使用二分查找法查找答案，根据每次查找的结果缩小答案所在范围，最后找出答案。

何时使用？

  • 与答案相关的变量有单调性
  • 即随着答案的变大，这个变量会单调递减 / 递增
  • 相当于我们在一个一次函数上二分寻找目标点
  • 出现最大值最小 / 最小值最大时可以考虑
int binary(int l, int r)
{
    int ans = 0;
    while (l <= r)
    {
        int mid = (l + r) >> 1;
        //此处是查找符合条件的最大值
        if (check(mid))
            ans = mid, r = mid + 1;
        else
            l = mid + 1;
        // 查找最小值
        // if(check(mid))
        //     ans = mid, r = mid + 1;
        // else
        //     l = mid + 1;
    }
    return ans;
}

# 三分

# 前提

三分搜索适用于具有凹凸性的函数，如图所示为一个具有凸性质的函数（注意不一定为一个标准的二次函数，只需要在唯一极值两端的单调性不同即可）在这个函数中我们就可以使用三分搜索。

  • 对于一个有极大值的单峰函数，当 f (lm)>f (rm) 时，只会有两种情况：

    1. lm 在对称轴左边，rm 在对称轴右边

    2. lm 和 rm 都在对称轴右边

  • 回想二次函数图像，一定可以舍掉右边的区间

  • < 的情况同理

  • 每次舍掉 1/3, 复杂度 O (2log3 (n)), 比二分要慢

const int eps = 1e-6;
double T_search(double l, double r)
{
    double lm = (l + l + r) / 3, rm = (l + r + r) / 3;
    if(f(lm)>f(rm)) r = rm;
    else l = lm;
}

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <math.h>
using namespace std;

int arr[10][10];
int x[10];
int sum;

int Gauss(int m, int n)//矩阵行数列数
{
	int row = 0, col = 0;
	while (row < m && col < n)
	{
		if (arr[row][col] == 0)
			for (int i = row + 1; i < m; i++)
				if (arr[i][col])
					for (int j = col; j <= n; j++)
						swap(arr[i][j], arr[row][j]);

		if (arr[row][col] == 0)
		{
			col++;
			continue;
		}

		for (int i = row + 1; i < m; i++) 
			if (arr[i][col])
			{
				//因为这里使用的是int型,所以消列首元素不一定为1
				int ta = arr[i][col];//被消列首元素大小
				int tb = arr[row][col];//消列首元素大小
				for (int j = col; j <= n; j++)
				{
					arr[i][j] = arr[i][j] * tb - arr[row][j] * ta;//使被消列首元素为0
					arr[i][j] = (arr[i][j] % 4 + 4) % 4;//使该数亦然在0-4范围内
				}
			}
		row++;
		col++;
	}

	for (int i = n - 1; i >= 0; i--)
	{
		int temp = arr[i][n];
		for (int j = i + 1; j < n; j++) 
			if (arr[i][j])
				temp -= arr[i][j] * x[j];
		temp = (temp % 4 + 4) % 4;

		//此处开始枚举
		for (x[i] = 0; x[i] <= 3; x[i]++)  // 转5次和转1次没区别，所以只有4种
			if ((x[i] * arr[i][i] % 4 + 4) % 4 == temp) // 依题意必定有解,所以找到解之后回代到上面去继续求其他未知数的值qwqq
				break;
		sum += x[i];
	}
	return 0;
}

int main()
{
	//    freopen("in.txt","r",stdin);

	arr[0][0] = arr[1][0] = arr[3][0] = arr[4][0] = 1;
	arr[0][1] = arr[1][1] = arr[2][1] = 1;
	arr[1][2] = arr[2][2] = arr[4][2] = arr[5][2] = 1;
	arr[0][3] = arr[3][3] = arr[6][3] = 1;
	arr[1][4] = arr[3][4] = arr[4][4] = arr[5][4] = arr[7][4] = 1;
	arr[2][5] = arr[5][5] = arr[8][5] = 1;
	arr[3][6] = arr[4][6] = arr[6][6] = arr[7][6] = 1;
	arr[6][7] = arr[7][7] = arr[8][7] = 1;
	arr[4][8] = arr[5][8] = arr[7][8] = arr[8][8] = 1;

	for (int i = 0; i < 9; i++)
	{
		scanf("%d", &arr[i][9]);
		arr[i][9] = (4 - arr[i][9]) % 4; // 将钟的位置转化为要转的次数
	}
	Gauss(9, 9);
	for (int j = 0; j < 9; j++)
		while (x[j])
		{
			printf("%d", j + 1);
			x[j]--;
			sum--;
			printf(sum > 0 ? " " : "\n");
		}
}

#include <iostream>
#include<algorithm>
#include <map>
#include <string>
using namespace std;
map<string, int> my;
int N, Map[10], restCard[10];
string ans;
char tab[7][20];
bool win;
void check()
{
    int eneny[] = { 0,1,2,3,4,5,6 };
    //cout<<n<<"  "<<endl;
    do {
        int i = 0, j = 0;
        while (i < N && j < N)
        {
            if (1 << Map[i] & restCard[eneny[j]]) i++;
            else j++;
        }
        if (i == N) return;
    } while (next_permutation(eneny, eneny + N));
    string tmp = "";
    for (int i = 0; i < N; i++)
    {
        if (i) tmp += " ";
        tmp += string(tab[Map[i]]);
    }
    win = true;
    if (ans == "" || ans > tmp) ans = tmp;
}
int main()
{
    int t, ncas = 1;
    scanf("%d", &t);
    for (; ncas <= t; ncas++)
    {
        char name[20];
        scanf("%d", &N);
        for (int i = 0; i < N; i++)
        {
            getchar();
            scanf("%s", name);
            strcpy(tab[i], name);
            my[string(name)] = i;
        }
        memset(restCard, 0, sizeof(restCard));
        for (int i = 0; i < N; i++)
        {
            int m;
            scanf("%d", &m);
            getchar();
            for (int j = 0; j < m; j++)
            {
                scanf("%s", name);
                restCard[i] |= 1 << my[string(name)];
            }
        }
        win = false;
        for (int i = 0; i < N; i++)
            Map[i] = i;
        ans = "";
        do {
            check();
        } while (next_permutation(Map, Map + N));
        printf("Case %d: ", ncas);
        puts(win ? "Yes" : "No");
        if (win)
            cout << ans << endl;
    }
    system("PAUSE");
    return 0;
}

<a name='gaosimuban'> 高斯消元模板 </a>

洛谷 p3389

#include <iostream>
#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;

const int MAXN = 1000;
const double eps = 1e-10;
void swap(double& a, double& b)
{
    double tmp = a;
    a = b;
    b = tmp;
}

int n;//矩阵的阶数
double arr[MAXN][MAXN];//矩阵及增广矩阵
double ans[MAXN];//答案向量
int gauss() {
    int r, c;//row, column
    for (r = 0, c = 0; c < n; c++)                              //现在在消第c列的元
    {                            
        int t = r;                                              //这里找到第c列r行及其之下中最大非0元素,将其行与目前行交换
        for (int i = r; i < n; i++)
            if (fabs(arr[i][c]) > fabs(arr[t][c])) t = i;
        if (fabs(arr[t][c]) < eps) continue;                    //此处判断是否该列第r行及其以下所有元素是否均为0,若均为0则不需要消元
        for (int i = c; i <= n; i++) swap(arr[t][i], arr[r][i]);//交换两行,这里包括增广矩阵的一列,所以为'<=n'
        for (int i = n; i >= c; i--) arr[r][i] /= arr[r][c];    //从后往前依此除以该行第一个非0元素,将该非0元素变成1

        //此处开始消元
        for (int i = r + 1; i < n; i++)                         //从第r+1行开始,下面每一行减去r行的对应倍数使首非0元素为0
            if (fabs(arr[i][c]) > eps)
                for (int j = n; j >= c; j--)
                    arr[i][j] -= arr[r][j] * arr[i][c];         //这里就是第i行减去 arr[i][c]*第r行 使得arr[i][c] -= arr[i][c] * arr[r][c] (arr[r][c]=1)

        r++;                                                    //切换到下一行
    }
    //此处消元结束，r即为矩阵的行秩，判断是否有解
    if (r < n)                                                  //行秩小于行数，判断增广矩阵中最后一列在r行之后是否有非0数
    {                                                           //若无则方程有无数解，若有则方程无解
        for (int i = r; i < n; i++)
            if (arr[i][n]) return 0;//无解
        return 1;//有无数解
    }
    //行满秩，m=n
    for (int i = n - 1; i >= 0; i--)                            //回代求解
        for (int j = i + 1; j < n; j++)                         
            arr[i][n] -= arr[i][j] * arr[j][n];
    for (int i = 0; i < n; i++)
        ans[i] = arr[i][n];
    return 2;//有唯一解
}
int main()
{
    scanf("%d", &n);
    for (int i = 0; i < n; i++)
        for (int j = 0; j <= n; j++)
            scanf("%lf", &arr[i][j]);
    int a = gauss();
    if (a != 2) cout << "No Solution" << endl;
    else
    {
        for (int i = 0; i < n; i++)
            printf("%.2lf\n", ans[i]);
    }
    return 0;
}
