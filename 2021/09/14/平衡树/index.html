# 平衡树

# 介绍

​ 平衡树是计算机科学中的一类数据结构，为改进的二叉查找树。一般的二叉查找树的查询复杂度取决于目标结点到树根的距离（即深度），因此当结点的深度普遍较大时，查询的均摊复杂度会上升。为了实现更高效的查询，产生了平衡树。

# 操作

  • 旋转（Rotate）：几乎所有平衡树的操作都基于树旋转操作（也有部分基于重构，如替罪羊树），通过旋转操作可以使得树趋于平衡。对一棵查找树（search tree）进行查询、新增、删除等动作，所花的时间与树的高度 h 成比例，并不与树的容量 n 成比例。如果可以让树维持平衡，也就是让 h 维持在O(logn)O(log\ n)O(log n) 的左右，就可以在的O(logn)O(log\ n)O(log n) 复杂度内完成各种基本操作。
  • 插入（insert)：在树中插入一个新值。
  • 删除（delete）：在树中删除一个值。
  • 查询前驱（predecessor）：前驱定义为小于 x，且最大的数。
  • 查询后继（successor）：后继定义为大于 x，且最小的数。
  • 在维护节点大小（size）后，可以支持以下操作：
  • 查询排名（rank）：排名定义为比 x 小的数的个数加一。
  • 查询第 k 大：即排名为 k 的数。

# 各种平衡树

# AVL 树

#include <iostream>
#include <cstdio>
#include <algorithm>
#include <set>
using namespace std;

//height为以该节点为根的树的高度
struct Node { int data; int height; Node* left; Node* right; };

int treeHeight(Node* root)	//查询以root为根节点的树的高度
{
	if (root == nullptr) return 0;
	return max(treeHeight(root->left), treeHeight(root->right)) + 1;
}

int treeBalanceFactor(Node* root)	//计算root节点的平衡因子
{
	if (root == nullptr) return 0;
	return (root->left ? root->left->height : 0) - (root->right ? root->right->height : 0);
}

Node* treeRotate(Node* root, bool left)	//树的旋转操作
{
	Node* newRoot = nullptr;
	if (left) {	//左旋与右旋相反
		newRoot = root->right;
		root->right = newRoot->left;
		newRoot->left = root;
	}
	else {	//右旋操作即为将现有根节点的左子节点变为新的根节点,左子节点的右子节点变为原根节点的左子节点
		newRoot = root->left;
		root->left = newRoot->right;
		newRoot->right = root;
	}
	//只有原根节点与现根节点的高度发生变化,子树所有节点高度不变
	//先对原根节点求高,再对新根节点求高(原根节点现在为新根节点的子节点
	root->height = max(treeHeight(root->left), treeHeight(root->right)) + 1;
	newRoot->height = max(treeHeight(newRoot->left), treeHeight(newRoot->right)) + 1;

	return newRoot;
}

Node* treeReBalance(Node* root)
{
	int factor = treeBalanceFactor(root);
	if (factor > 1) {	//左子树超重,需要调整
		if (treeBalanceFactor(root->left) < 0)	//LR型,左子节点的右子树超重,先调整左子节点,将其转化为LL型
			root->left = treeRotate(root->left, true);
		return treeRotate(root, false);	//LL型,对根节点做一次右旋即可
	}
	if (factor < -1) {	//右子树超重
		if (treeBalanceFactor(root->right) > 0)	//RL型,右子节点的左子树超重,先调整右子节点,将其转化为RR
			root->right = treeRotate(root->right, false);
		return treeRotate(root, true);	//RR型,进行一次左旋
	}
	return root;
}

void treeInsert(Node*& root, int value)
{
	if(root == nullptr) {	//递归到插入处
		root = static_cast<Node*>(malloc(sizeof(Node)));
		root->data = value;
		root->left = root->right = nullptr;
	}else if (root->data == value) return;
	else {
		if (root->data < value) treeInsert(root->right, value);
		else treeInsert(root->left, value);
	}
	//递归到最底层后返回更新节点高度,并平衡节点
	root->height = treeHeight(root);
	root = treeReBalance(root);
}

void treeDelete(Node*& root, int value)
{
	if(root) {
		if (root->data > value) treeDelete(root->left, value);
		else if (root->data < value) treeDelete(root->right, value);
		else{
			if(root->right&&root->left) {
				//左右孩子均存在,找右子树的最小节点作为继任者
				//将最小节点的值与要删除的节点替换,最小节点的父节点的左节点修改为最小节点的右节点
				//最后删除最小节点
				Node* successor = root->right, * sucFather = root;
				while (successor->left) {
					sucFather = successor;
					successor = successor->left;
				}
				root->data = successor->data;
				sucFather->left = successor->right;
				free(successor);
			}else {
				//左右孩子不都存在,将该节点的父节点与它的左/右子节点相连
				//位置与它与父节点连接的位置相同
				//删除该节点
				Node* freeNode = root;
				root = root->left ? root->left : root->right;
				free(freeNode);
			}
		}
		//当root不为空时,回溯跟新高度,并重新平衡子树
		if(root) root->height = treeHeight(root);
		root = treeReBalance(root);
	}
}

int main()
{
	//freopen("in", "r", stdin);
	Node* root = nullptr;
	treeInsert(root, 1);
	treeInsert(root, 4);
	treeInsert(root, 6);
	treeInsert(root, 7);
	treeInsert(root, 3);
	treeInsert(root, 14);
	treeInsert(root, 5);
	treeInsert(root, 8);
	treeDelete(root, 7);
	treeDelete(root, 5);
	treeDelete(root, 14);
	treeInsert(root, 71);
	treeInsert(root, 54);
	treeInsert(root, 33);
	treeInsert(root, 18);
	treeInsert(root, 90);
	treeInsert(root, 91);
	treeDelete(root, 4);
	treeDelete(root, 90);
	return 0;
}
