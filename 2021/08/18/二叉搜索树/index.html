# 二叉搜索树

洛谷 P5076

BST（Binary Search Tree），二叉搜索树，又叫二叉排序树

是一棵空树或具有以下几种性质的树:

  1. 若左子树不空，则左子树上所有结点的值均小于它的根结点的值
  2. 若右子树不空，则右子树上所有结点的值均大于它的根结点的值
  3. 左、右子树也分别为二叉排序树
  4. 没有权值相等的结点。

在数据中遇到多个相等的数该怎么办呢，显然我们可以多加一个计数器，就是当前这个值出现了几遍。

那么我们的每一个节点都包含以下几个信息:

  1. 当前节点的权值，也就是序列里的数
  2. 左孩子的下标和右孩子的下标，如果没有则为 0
  3. 计数器，代表当前的值出现了几遍
  4. 子树大小和自己的大小的和
struct node {int v, l, r, cnt, size;} tree[100000];
// 结点的值，左、右孩子的下标，值出现的次数，子树大小和自己大小的和

使用 c++ 递归实现:

#include <algorithm>
#include <cmath>
#include <cstdio>
#include <iostream>
const int INF = 2147483647;

struct node {
    int v, l, r, cnt, size;
} tree[100000];     //节点的值,左、右孩子的下标,值出现的次数,子树大小和自己大小的和
int count;      //树节点总数

void add(int x, int v)
{                           //x代表当前递归到的节点下标,v是要插入的值
    tree[x].size++;         //增加以这个节点为根的子树大小
    if (tree[x].v == v) {   //相等,则在该节点上数量加一
        tree[x].cnt++;
        return;
    }
    if (tree[x].v > v) {    //v在节点x的左子树里
        if (tree[x].l != 0) //若左子树不为空,就递归进入左子树
            add(tree[x].l, v);
        else {              //否则,v就是x的左子树的值
            count++;        //count是当前树的节点个数,从1开始
            tree[count].v = v;
            tree[count].size = tree[count].cnt = 1;
            tree[x].l = count;
        }
    } else {                //右子树同理
        if (tree[x].r != 0)
            add(tree[x].r, v);
        else {
            count++;
            tree[count].v = v;
            tree[count].size = tree[count].cnt = 1;
            tree[x].r = count;
        }
    }
}
int query_pre(int x, int v, int ans)
{                               //x代表当前查找到的节点下标,v是要查找前驱的值,ans是目前找到的比v小的数的最大值
    if (tree[x].v >= v) {       //当前值大于v,说明查找的数过大,往左子树查找
        if (tree[x].l == 0)     //若没有左子树,则直接返回ans
            return ans;
        else                    //存在左子树,去左子树中查找
            return query_pre(tree[x].l, v, ans);
    } else {                    //当前值小于v,检验是否为最大值
        if (tree[x].r == 0)     //没有右子树,直接返回该节点的值
            return tree[x].v;
        if (tree[x].cnt != 0)   //防止被删除的数据被计入答案,更新ans的值为当前节点的值
            return query_pre(tree[x].r, v, tree[x].v);
        else                    //否则不需要更新ans
            return query_pre(tree[x].r, v, ans);
    }
}
int query_suc(int x, int v, int ans)
{
    if (tree[x].v <= v) {
        if (tree[x].r == 0)
            return ans;
        else
            return query_suc(tree[x].r, v, ans);
    } else {
        if (tree[x].l == 0)
            return tree[x].v;
        if (tree[x].cnt != 0)
            return query_suc(tree[x].l, v, tree[x].v);
        else
            return query_suc(tree[x].l, v, ans);
    }
}
int query_val(int x, int v)
{                                       //返回比v小的数的个数,输出排名时+1
    if (x == 0)                         //无比它小的数
        return 0;
    if (tree[x].v == v)
        return tree[tree[x].l].size;    //返回该结点左子树的大小
    if (tree[x].v > v)                  //结点v大于v,去结点的左子树查找
        return query_val(tree[x].l, v);
    else                                //结点v小于v,去右子树查找,并加上该节点左子树大小和该节点大小
        return query_val(tree[x].r, v) + tree[tree[x].l].size + tree[x].cnt;
}
int query_rnk(int x, int rank)
{
    if (x == 0)
        return INF;
    if (tree[tree[x].l].size >= rank) 	//左子树大小大于rank,则答案在左子树里
        return query_rnk(tree[x].l, rank);
    if (tree[tree[x].l].size + tree[x].cnt >= rank) //左子树大小加上当前节点的值的个数大于rank时,说明当前节点的值就是要求的答案
        return tree[x].v;
    else 								//说明答案在右子树中,去右子树中查找的时候rank需要减去左子树大小与当前节点的值的个数,因为他们的排名都比右子树中的值的排名小
        return query_rnk(tree[x].r, rank - tree[tree[x].l].size - tree[x].cnt);
}

int main()
{
    //freopen("in", "r", stdin);
    int n;
    scanf("%d", &n);
    while (n--) {
        int opt, x, ans = -1;
        scanf("%d%d", &opt, &x);
        switch (opt) {
        case 1:
			printf("%d\n", query_val(1, x) + 1);
            break;
        case 2:
			printf("%d\n", query_rnk(1, x));
            break;
        case 3:
			printf("%d\n", query_pre(1, x, -INF));
            break;
        case 4:
            printf("%d\n", query_suc(1, x, INF));
            break;
        case 5:
            if (count == 0) {
                tree[1].cnt = tree[1].size = 1;
                tree[1].v = x;
                count = 1;
            } else {
                add(1, x);
            }
        }
    }
    return 0;
}
