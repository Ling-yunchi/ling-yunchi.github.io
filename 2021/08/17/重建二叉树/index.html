# 重建二叉树

# 洛谷 P1827

# 题目描述

农夫约翰非常认真地对待他的奶牛们的血统。然而他不是一个真正优秀的记帐员。他把他的奶牛 们的家谱作成二叉树，并且把二叉树以更线性的 “树的中序遍历” 和 “树的前序遍历” 的符号加以记录而 不是用图形的方法。

你的任务是在被给予奶牛家谱的 “树中序遍历” 和 “树前序遍历” 的符号后，创建奶牛家谱的 “树的 后序遍历” 的符号。每一头奶牛的姓名被译为一个唯一的字母。（你可能已经知道你可以在知道树的两 种遍历以后可以经常地重建这棵树。）显然，这里的树不会有多于 26 个的顶点。 这是在样例输入和 样例输出中的树的图形表达方式：

　　　　　　　　 C
　　　　　　   /  \
　　　　　　  /　　\
　　　　　　 B　　  G
　　　　　　/ \　　/
　　　　   A   D  H
　　　　　　  / \
　　　　　　 E   F

树的中序遍历是按照左子树，根，右子树的顺序访问节点。

树的前序遍历是按照根，左子树，右子树的顺序访问节点。

树的后序遍历是按照左子树，右子树，根的顺序访问节点。

# 输入格式

第一行： 树的中序遍历

第二行： 同样的树的前序遍历

# 输出格式

单独的一行表示该树的后序遍历。

树的前中后序遍历实际上就是访问根节点的前中后序，左子树右子树访问顺序不变。

英文缩写规律：

LR不变 前序中根节点为V 中后序中根节点为D
前： VLR
中： LDR
后： LRD

　　　　　　　　 C
　　　　　　   /  \
　　　　　　  /　　 \
　　　　　　 B　　   G
　　　　　　/ \　　 /
　　　　  A    D  H
　　　　　    / \
　　　　　   E   F
该树的前序遍历为：CBADEFGH 
中序遍历为：ABEDFCHG
后序遍历为：AEFDBHGC

前序遍历的特点是根结点在前，这可以帮我们找到根节点。

中序遍历的特点是根结点根节点左侧为左子树结点，右侧为右子树结点。

对于上面这个例子：

  • 首先在前序遍历找到整个树的根节点 C
  • 在中序遍历中找到 C，那么 ABEDF 在 C 的左子树上，HG 在 C 的右子树上
  • 重建左子树，在前序遍历中找到左子树的根节点为 B
  • 在中序遍历中左子树部分找到 B，那么 A 在 B 的左子树上，EDF 在右子树上
  • 在前序遍历中找到 B 的左子树的根节点 A，在中序遍历中发现 A 没有子节点，那么 A 为叶子结点。
  • 在前序遍历中找到 B 的右子树 EDF 的根节点 D，在中序遍历中找到 D 的左子树上的节点为 E，右子树上结点为 F
  • 。。。
  • 查找 C 的右子树的根节点 G，发现 G 的左子树上的结点为 H
  • 。。。
  • 重建完毕ヾ (・ω・`) o

上面这个步骤一层套一层的，把整个前序遍历遍历了一遍（听君一席话，如听一席话

我们可以使用递归来完成代码的编写：

#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cctype>
#include <cstring>
using namespace std;
char VLR[30], LDR[30];
int n = 0, cnt = 0;

struct node { int l, r; } tree[30];
inline int ctn(char a) { return a - 64; }
inline char ntc(int n) { return n + 64; }

void rebuild(int father, int l, int r, bool left)
{
	if (l > r)	// 这里不注意会导致数组越界qwqqqqq
		return;
	int root = ctn(VLR[cnt++]);
	if (left)
		tree[father].l = root;
	else
		tree[father].r = root;
	if (l == r)
		return;
	for (int i = l; i <= r; i++)
		if (LDR[i] == ntc(root)) {
			rebuild(root, l, i - 1, true);
			rebuild(root, i + 1, r, false);
			break;
		}
}

void LRD(int root)
{
	if (tree[root].l)
		LRD(tree[root].l);
	if (tree[root].r)
		LRD(tree[root].r);
	printf("%c", ntc(root));
}

int main() {
	//freopen("in", "r", stdin);
	scanf("%s", LDR);
	scanf("%s", VLR);
	while (isalpha(VLR[++n]));
	int root = ctn(VLR[cnt++]);
	for (int i = 0; i <= n - 1; i++)
		if (LDR[i] == ntc(root)) {
			rebuild(root, 0, i - 1, true);
			rebuild(root, i + 1, n - 1, false);
			break;
		}

	LRD(root);

	return 0;
}

# 洛谷 P1030

# 题目描述

给出一棵二叉树的中序与后序排列。求出它的先序排列。（约定树结点用不同的大写字母表示，长度≤8\le 8≤8）。

# 输入格式

2 行，均为大写字母组成的字符串，表示一棵二叉树的中序与后序排列。

# 输出格式

1 行，表示一棵二叉树的先序。

老方法：

  • 在后序排列中从后往前遍历根节点
  • 在中序排列中找到该根节点
  • 递归左右建树
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cctype>
#include <cstring>
using namespace std;
char LRD[30], LDR[30];
int n = 0, cnt = 0;

struct node { int l, r; } tree[30];
inline int ctn(char a) { return a - 64; }
inline char ntc(int n) { return n + 64; }

void rebuild(int father, int l, int r, bool left)
{
	if (l > r) return;
	int root = ctn(LRD[cnt--]);
	if (left) tree[father].l = root;
	else tree[father].r = root;
	if (l == r) return;
	for (int i = l; i <= r; i++)
		if (LDR[i] == ntc(root)) {
			rebuild(root, i + 1, r, false);
			rebuild(root, l, i - 1, true);
			break;
		}
}

void VLR(int root)
{
	printf("%c", ntc(root));
	if (tree[root].l) VLR(tree[root].l);
	if (tree[root].r) VLR(tree[root].r);
}

int main() {
	//freopen("in", "r", stdin);
	scanf("%s", LDR);
	scanf("%s", LRD);
	while (isalpha(LRD[++n]));
	cnt = n - 1;
	int root = ctn(LRD[cnt--]);
	for (int i = 0; i <= n - 1; i++)
		if (LDR[i] == ntc(root)) {
            //注意LRD从后往前遍历需要先建右子树再建左子树，此处和rebuild中的建树顺序都需要改变
			rebuild(root, i + 1, n - 1, false);
			rebuild(root, 0, i - 1, true);
			break;
		}
	VLR(root);
	return 0;
}

# 不建树输出前、后序遍历

是不是觉得建树也很麻烦，码代码时出错就是调试地狱 qwqq

还有不建树就能完成上面两道题的方法

实际上我们可以从中序与前 / 后序遍历中得知整个树的结构

于是实际上不需要建树，我们就能按照树的规律遍历

方法核心为使用 DFS 遍历到底后返回时输出

# P1827

#include <cstdio>
#include <iostream>
#include <cctype>

using namespace std;
char VLR[30], LDR[30];
int n = 0, cnt = 0;

void DFS(int l, int r)
{
	if(l > r) return;
	int root = cnt++;
	for (int i = 0; i < n; i++)
		if (VLR[root] == LDR[i]) {
			DFS(l, i - 1);
			DFS(i + 1, r);
			printf("%c", VLR[root]);
			break;
		}
}

int main() {
	freopen("in", "r", stdin);
	scanf("%s", LDR); scanf("%s", VLR);
	while (isalpha(VLR[++n]));

	int root = cnt++;
	for(int i=0;i<n;i++)
		if(VLR[root]==LDR[i]) {
			DFS(0, i - 1);
			DFS(i + 1, n - 1);
			printf("%c", VLR[root]);
			break;
		}

	return 0;
}

# P1030

#include <cstdio>
#include <iostream>
#include <cctype>

using namespace std;
char LRD[30], LDR[30];
int n = 0;

void DFS(int l, int r)
{
	if(l > r) return;
	//在后续遍历中找到子树的根,为子树中元素出现在后序遍历中最晚的元素
	int root = -1;
	for (int i = l; i <= r; i++)
		for (int j = 0; j < n; j++)
			if (LDR[i] == LRD[j] && j > root)
				root = j;
	for (int i = 0; i < n; i++)
		if (LRD[root] == LDR[i]) {
			printf("%c", LRD[root]);
			DFS(l, i - 1);
			DFS(i + 1, r);
			break;
		}
}

int main() {
	freopen("in", "r", stdin);
	scanf("%s", LDR); scanf("%s", LRD);
	while (isalpha(LRD[++n]));

	int root = n - 1;
	for(int i=0;i<n;i++)
		if(LRD[root]==LDR[i]) {
			printf("%c", LRD[root]);
			DFS(0, i - 1);
			DFS(i + 1, n - 1);
			break;
		}

	return 0;
}
